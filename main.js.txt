import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js';

// 1. Initialisation de la Scène, Caméra et Rendu
const scene = new THREE.Scene();
scene.background = new THREE.Color(0x1a1a1a); // Fond gris foncé

const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
camera.position.set(0, 4, 8); // Ajuste ces chiffres si tu apparais trop loin/près

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.shadowMap.enabled = true;
document.getElementById('canvas-container').appendChild(renderer.domElement);

// 2. Contrôles de déplacement (Souris : rotation / Molette : zoom / Clic droit : translation)
const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true; 
controls.dampingFactor = 0.05;

// 3. Lumières de secours (au cas où ton export Blender soit trop sombre)
const ambientLight = new THREE.AmbientLight(0xffffff, 0.7);
scene.add(ambientLight);

const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);
dirLight.position.set(5, 10, 5);
scene.add(dirLight);

// 4. TA BASE DE DONNÉES (Remplis-la au fur et à mesure avec les noms trouvés grâce à l'alerte !)
const equipmentsData = {
    // Exemple : si l'alerte te dit que ton piano s'appelle "Cube.002"
    "Cube.002": { 
        title: "Mon super Piano MIDI", 
        desc: "C'est le clavier maître qui me sert à composer toutes mes mélodies." 
    },
    // Exemple : si ton canapé s'appelle "Sofa_Mesh"
    "Sofa_Mesh": { 
        title: "Le Canapé Chill", 
        desc: "Parfait pour écouter les mixages ou faire une pause." 
    }
};

// 5. Chargement de ton fichier Blender studio.glb
const loader = new GLTFLoader();
let selectableObjects = [];

loader.load(
    'studio.glb', 
    (gltf) => {
        const model = gltf.scene;
        scene.add(model);
        
        // On centre le modèle automatiquement
        model.position.set(0, 0, 0);

        // On liste tous les sous-éléments du modèle pour pouvoir cliquer dessus
        model.traverse((child) => {
            if (child.isMesh) {
                selectableObjects.push(child);
            }
        });
        console.log("Modèle 3D chargé avec succès !");
    },
    (xhr) => {
        console.log((xhr.loaded / xhr.total * 100) + '% chargé');
    },
    (error) => {
        console.error("Erreur lors du chargement du fichier GLB :", error);
    }
);

// 6. Gestion du Clic et Technique de l'Espion
const raycaster = new THREE.Raycaster();
const mouse = new THREE.Vector2();

window.addEventListener('click', (event) => {
    // Évite de déclencher un clic si on clique sur l'interface d'information
    if (event.target.closest('#info-box')) return;

    // Calcul des coordonnées de la souris
    mouse.x = (event.clientX / window.innerWidth) * 2 - 1;
    mouse.y = -(event.clientY / window.innerHeight) * 2 + 1;

    raycaster.setFromCamera(mouse, camera);
    const intersects = raycaster.intersectObjects(selectableObjects);

    if (intersects.length > 0) {
        let hitObject = intersects[0].object;

        // 🚨 TECHNIQUE DE L'ESPION : Une alerte va s'ouvrir pour te donner le nom de l'objet cliqué !
        alert("Tu as cliqué sur l'objet nommé : " + hitObject.name);
        console.log("Nom complet de l'objet :", hitObject.name);

        // Recherche si cet objet ou l'un de ses parents est enregistré dans ta base de données
        let foundData = null;
        let current = hitObject;
        
        while (current) {
            if (equipmentsData[current.name]) {
                foundData = equipmentsData[current.name];
                break;
            }
            current = current.parent;
        }

        // Si l'objet est connu dans ta base de données, on affiche sa fiche info
        if (foundData) {
            showInfo(foundData.title, foundData.desc);
        }
    }
});

// 7. Fonctions pour la boîte d'infos
const infoBox = document.getElementById('info-box');
function showInfo(title, text) {
    document.getElementById('info-title').innerText = title;
    document.getElementById('info-description').innerText = text;
    infoBox.style.display = 'block';
}

document.getElementById('close-btn').addEventListener('click', () => {
    infoBox.style.display = 'none';
});

// 8. Boucle de rendu (Animation)
function animate() {
    requestAnimationFrame(animate);
    controls.update();
    renderer.render(scene, camera);
}
animate();

// Redimensionnement automatique si on change la taille de la fenêtre
window.addEventListener('resize', () => {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
});
